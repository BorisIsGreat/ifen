// Load the Google API client library require_once __DIR__ . '/vendor/autoload.php'; // Set the credentials for the Google API client $client = new Google_Client(); $client->setAuthConfig('path/to/client_secret.json'); $client->setAccessType('offline'); // Authenticate with Google $service = new Google_Service_Sheets($client); // Set the spreadsheet ID and range where you want to write the data $spreadsheetId = '1ufA3qXALuBWyHFqe3c9KCsnIUFq-Nw-cxne3F6q4ixk/edit#gid=0'; $range = 'Sheet1!A2'; // Set the value you want to write to the cell $value = 'Moien'; // Build the request to write the data to the cell $requestBody = new Google_Service_Sheets_ValueRange([ 'range' => $range, 'majorDimension' => 'ROWS', 'values' => [[$value]], ]); // Make the API request to write the data to the cell $params = [ 'valueInputOption' => 'RAW' ]; $response = $service->spreadsheets_values->update($spreadsheetId, $range, $requestBody, $params); // Output a message indicating whether the data was written successfully if ($response->getUpdatedCells() > 0) { echo 'Data written to Google Sheets!'; } else { echo 'Failed to write data to Google Sheets.'; }
